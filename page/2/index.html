func SetTitle() 
    "如果文件类型为.sh文件 
    if &filetype == 'sh'
        call setline(1,"\#########################################################################") 
        call append(line("."), "\# File Name: ".expand("%")) 
        call append(line(".")+1, "\# Author: will") 
        call append(line(".")+2, "\# mail: 457383152@qq.com") 
        call append(line(".")+3, "\# Created Time: ".strftime("%c")) 
        call append(line(".")+4, "\#########################################################################") 
        call append(line(".")+5, "\#!/bin/bash") 
        call append(line(".")+6, "")
    elseif &filetype == 'python'
        call setline(1,"\#!/usr/bin/env python")
        call append(line("."),"\#encoding:utf-8")
        call append(line(".")+1,"\#########################################################################")
        call append(line(".")+2, "\# File Name: ".expand("%"))
        call append(line(".")+3, "\# Author: will")
        call append(line(".")+4, "\# mail: 457383152@qq.com")
        call append(line(".")+5, "\# Created Time: ".strftime("%c"))
        call append(line(".")+6, "\#########################################################################")
        call append(line(".")+7, "")

    else 
        call setline(1, "/*************************************************************************") 
        call append(line("."), "    > File Name: ".expand("%")) 
        call append(line(".")+1, "  > Author: will") 
        call append(line(".")+2, "  > Mail: 457383152@qq.com ") 
        call append(line(".")+3, "  > Created Time: ".strftime("%c")) 
        call append(line(".")+4, " ************************************************************************/") 
        call append(line(".")+5, "")
    endif
    if &filetype == 'cpp'
        call append(line(".")+6, "#include")
        call append(line(".")+7, "using namespace std;")
        call append(line(".")+8, "")                                                                                                                                                                         
    endif
    if &filetype == 'c'
        call append(line(".")+6, "#include")
        call append(line(".")+7, "")
    endif
    if &filetype == 'python'
        call append(line(".")+8, "import sys")
        call append(line(".")+9, "import os")
    endif

endfunc

移动端DSP系统介绍

DSP:广告主服务平台。主要负责根据广告主设定的目标受众、条件定向、广告出价等条件进行广告的投放及效果追踪。

DSP系统的主要模块：实时竞价模块、投放控制模块、效果追踪模块、出价模块、CTR预测模块、流量分析模块。
实时竞价模块：负责参与AD Exchange的实时竞价。具体功能为：判断AD Exchange发过来的广告位是否符合广告主的需求，进而决定是否进行竞价。整个过程需要在200ms内完成（Doubleclick需要在100ms内完成）。
投放控制模块：统计每个活动的实时出价情况及投放情况，然后根据运营和广告主设定的条件控制竞价模块的竞价速率。同时需要提供简易的报表给运营。
效果追踪模块：负责与广告主服务器或者第三方检测进行对接。接收他们的回调通知或回调他们以通知点击、安装。
CTR预测模块 ：负责对每个广告位的CTR进行预测，以供出价模块进行出价。
出价模块 ：根据CTR预测模块提供的CTR预测，计算出合理的价格提供给竞价模块。
流量分析模块：负责统计每个不同AD Exchange的流量分布。
